PiToTheta = function(Pis)
{
      if (abs(sum(Pis)-1)>1e-7) {print("Please use proper probability.")}
      pi0 = Pis[1]
      pis = Pis[-1]
      K = length(pis)
      Y = matrix(NA,nrow=K,ncol=1)
      for (s in 1:K)
      {
            Y[s,1] = log(pis[s]/pi0/choose(K,s))
      }
      M = matrix(0,nrow=K,ncol=K)
      for (i in 1:K)
      {
            for (j in 1:i)
            {
                  M[i,j] = choose(i,j)
            }
      }
      theta = solve(M,Y)
      theta
}
